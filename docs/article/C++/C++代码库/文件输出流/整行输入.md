#include<cstring>

using namespace std;
int main()
{
	/*Way1.数组字符串存储字符串(1)*/
	/*
	    用法：配合cin输入
	    缺点：1.数组长度固定
              2.不能存空格
	*/
	char name[]="我爱你5";
	char temp[]="123456";
	cout<<"name数组内容:"<<name<<endl<<"更新name数组为："<<endl;
	//cin输入，相当于把name数组清空了再输入
	cin>>name;              	   
	cout<<"更新后内容:"<<name<<endl<<endl;
	

	
	/*Way2.数组字符串存储字符串(2)*/
	/*
	    用法：1.配合getline输入
			  2.getline第一个参数是用来存储输入行的数组的名称,第二个参数是要读取的字符数+1。（算上结束符）
			  3.cin.getline()读取换行符，并把换行符换成结束符
	    缺点：1.数组长度固定
	*/
	
	//这个cin.get()可以读取任何字符，用来处理残留在流中的换行符,保证能接收cin.getline()的输入
	cin.get();
	cout<<"更新name数组为："<<endl;
    cin.getline(name,20);
	cout<<"更新后内容:"<<name<<endl<<endl;
    
    /*Way3.数组字符串存储字符串(3)*/
    /*
	    用法：1.配合get()输入
			  2.语法同上getline()
			  3.get()和getline()的区别在于，get()并不再读取并丢弃换行符，而是将其留在输入队列(流)中*
	    缺点：1.数组长度固定
	    
	    说明：1.因为前面的getline()读取了换行符。所以下面的cin.get()可以直接读取
              2.当get( )读取空行后，将设置失效位（failbit），这意味着接下来的输入将被阻断，可用cin.clear()恢复。
              3.51行必须加cin.get(),吃掉换行
	*/

    cout<<"更新name数组为："<<endl;
    cin.get(name,20);
    cout<<name<<endl;
    
	

    /*Way4.string的getline*/
    /*
	    用法：1.配合getline()输入
			  2.语法同上getline(cin,str)
			  3.读取并丢弃换行符
			  4.str长度任意
	*/
	cin.get();
	string str;
	getline(cin,str);
	cout<<"str的内容是:"<<str<<endl;
	



	/*注
	
	1.为什么要使用get( )，而不是getline( )呢？首先，老式实现没有getline( )。
	其次，get( )使输入更仔细。例如，假设用get( )将一行读入数组中。
	如何知道停止读取的原因是由于已经读取了整行，而不是由于数组已填满呢？
	查看下一个输入字符，如果是换行符，说明已读取了整行；否则，说明该行中还有其他输入。
	
	2.输入字符串可能比分配的空间长。
	如果输入行包含的字符数比指定的多
	则getline( )和get( )将把余下的字符留在输入队列中
	而getline( )还会设置失效位，并关闭后面的输入。
	
	*/
	
}
