#include<iostream>
using namespace std;
int arry1[500005],arry2[500005];// is1为原数组，is2为临时数组，n为个人定义的长度
long merge(int low,int mid,int high);
long mergeSort(int a,int b);
int main()
{
    int n;
    while(cin>>n)
    {   if(n==0) break;
        long long count;			//坑在这里 
        count=0;
        for(int i=0;i<n;i++)
        cin>>arry1[i];
        count=mergeSort(0,n-1);
        cout<<count<<endl;
    }
     
    return 0;
     
}
 
 
 
long merge(int low,int mid,int high)
{
    int i=low,j=mid+1,k=low;				
    long long count=0;
    
    //合并操作 
    while(i<=mid&&j<=high)					//i和j都只会加，保证不越 "数组" 
    if(arry1[i]<=arry1[j])                  //前数列的首元素较小，不用交换，不计数 
        arry2[k++]=arry1[i++];
    else
    {
        arry2[k++]=arry1[j++];				//后数列的首元素较小 
        count+=j-k;                      	//记录跨了多少步
    }
     
    //某一个"数组" "空了"
    
    //如果前数组没空 
    while(i<=mid)
        arry2[k++]=arry1[i++];
         
    //如果后数组没空     
    while(j<=high)
        arry2[k++]=arry1[j++];
         
         
    for(i=low;i<=high;i++)// 写回原数组
        arry1[i]=arry2[i];
         
    return count;
}
 
long mergeSort(int a,int b)
{	//分治 
    if (a >= b)                             //注意有等于号，当分得只剩一个元素（1/2==0）或者没有元素，返回
        return 0;
     
        int mid=(a+b)/2;
        long long count=0;
        count+=mergeSort(a,mid);            //分组
        count+=mergeSort(mid+1,b);          //分组
        count+=merge(a,mid,b);              //调用merge
         
        return count;
     
}
