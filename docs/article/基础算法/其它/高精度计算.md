### 高精度加法

```c++
// C = A + B, A >= 0, B >= 0
//vector 的尾部存高位
vector<int> add(vector<int> &A, vector<int> &B)
{
    if (A.size() < B.size()) return add(B, A);

    vector<int> C;
    int t = 0;
    for (int i = 0; i < A.size(); i ++ )
    {
        t += A[i];
        if (i < B.size()) t += B[i];
        C.push_back(t % 10);
        t /= 10;
    }

    if (t) C.push_back(t);
    return C;
}
```

### 高精度减法

```c++
// C = A - B, 满足A >= B, A >= 0, B >= 0
// 其它情况可以延申处理
vector<int> sub(vector<int> &A, vector<int> &B)
{
    vector<int> C;
    //每次循环开始时，t是记录低位的借位情况
    for (int i = 0, t = 0; i < A.size(); i ++ )
    {
        t = A[i] - t;
        if (i < B.size()) t -= B[i];
        //让t变正数
        C.push_back((t + 10) % 10);
        //处理借位
        if (t < 0) t = 1;
        else t = 0;
    }

    //把减为0的高位弹出
    while (C.size() > 1 && C.back() == 0) C.pop_back();
    return C;
}


//配合比较大小的函数
bool cmp(vector<int> &A, vector<int> &B)
{
    if (A.size() != B.size()) return A.size() > B.size();

    for (int i = A.size() - 1; i >= 0; i -- )
        if (A[i] != B[i])
            return A[i] > B[i];

    return true;
}
```



### 高精度乘低精度

```c++
// C = A * b, A >= 0, b > 0
vector<int> mul(vector<int> &A, int b)
{
    vector<int> C;
    int t = 0;
    for (int i = 0; i < A.size() || t; i ++ )
    {
        if (i < A.size()) t += A[i] * b;
        C.push_back(t % 10);
        t /= 10;
    }

    return C;
}
```



### 高精度除以低精度 

```c++
// A / b = C ... r, A >= 0, b > 0
vector<int> div(vector<int> &A, int b, int &r)
{
    //C存结果
    //A高位在右
    vector<int> C;
    r = 0;
    for (int i = A.size() - 1; i >= 0; i -- )
    {
        r = r * 10 + A[i];
        C.push_back(r / b);
        //此处r代表借位
        r %= b;
    }
    
    //转向，让高位在右边
    reverse(C.begin(), C.end());
    //高位弹出
    while (C.size() > 1 && C.back() == 0) C.pop_back();
    return C;
}

```
