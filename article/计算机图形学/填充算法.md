## 代码

```python
from matplotlib import pyplot as plt


#给区间[lx,ly]的扫描线段上色
#边界不上色
def draw(lx,rx,y):
    xx = list(range(int(lx)+1,int(rx+1)-1))
    yy = [y for i in range(len(xx))]
    if y == 50:
        print("这一行的点对：",xx,yy)
    plt.scatter(xx, yy, color='red', marker='.')

#Aet结点结构
class AET_node:
    def __init__(self,px,delx,ymax,next = None):
        self.line = {}
        self.line['px'] = px
        self.line['del_x'] = delx
        self.line['y_max'] = ymax
        self.next = next


#输入图形，按顺时针输入顶点输
# p_set = [[2,7],[5,5],[11,8],[11,3],[5,1],[2,2]]
# p_set = [[100,100],[500,100],[500,10],[100,10]]
# p_set = [[-5,-5],[10,10]]
p_set = [[100,100],[500,100],[800,10],[100,-10]]
#p_set= [[50,50],[100,120],[150,80],[200,100],[220,0],[50,0],[100,50]]


#ml保存扫描线下界,ML保持上界,先给个初始值
ml = 100000000
Ml = -10000000


#Net表用字典
Net = {}
# 初始化Net表
# 在这个过程构造AET结点
for i in range(len(p_set)):
    #取出点
    p = p_set[i]
    #取相邻的点，因为按最输入的规定，相邻的点构成直线
    nxt_p = p_set[(i+1)%len(p_set)]

    #让直线方向指向 y值小的对应的点,顺便求斜率的倒数
    if p[1] >= nxt_p[1]:
        x = nxt_p[0]
        if nxt_p[1]-p[1]!=0:
            del_x = (nxt_p[0]-p[0])/(nxt_p[1]-p[1])
        else:
            del_x = 0
    else :
        x = p[0]
        if p[1] - nxt_p[1]!=0:
            del_x =  (p[0] - nxt_p[0])/(p[1] - nxt_p[1])
        else:
            del_x = 0

    y_max = max(nxt_p[1],p[1])
    new_Aet = AET_node(x,del_x,y_max)

    y_min = min(nxt_p[1], p[1])
    #将数据存进Net表
    #如果不存在这一列
    if Net.get(y_min) == None:
        Net[y_min] = new_Aet
    else:
        ptr = Net[y_min]
        while ptr.next != None:
            ptr = ptr.next
        ptr.next = new_Aet

    #更新扫描线边界
    ml = min(ml,y_min)
    Ml = max(Ml,y_max)




#新建空AET链,给这个链添加一个虚头节点。
AET = AET_node(-99999999,0,0)

for i in range(ml,Ml+1):
    #如果有这个点
    if Net.get(i)!=None:
        #取链
        ptr = Net[i]
        #插入排序AET(假定AET有序) O(n)
        #将数据放进一个列表,并排序
        tmp_l = []
        aet = AET.next
        while aet != None:
            tmp_l .append([aet.line['px'],aet.line['del_x'],aet.line['y_max']])
            aet = aet.next
        while ptr != None:
            tmp_l .append([ptr.line['px'],ptr.line['del_x'],ptr.line['y_max']])
            ptr = ptr.next
        tmp_l.sort()
        #把数据装回去AET
        AET = AET_node(-99999999, 0, 0)
        AET_ptr = AET
        for k in range(len(tmp_l)):
            item = tmp_l[k]
            AET_ptr.next = AET_node(item[0],item[1],item[2])
            AET_ptr = AET_ptr.next



    #遍历AET表，并把配对的区间
    ptr = AET.next

    #把AET的x存进列表
    tmp_x = []
    while ptr != None:
        tmp_x.append([ptr.line['px'],ptr.line['y_max']])
        ptr = ptr.next

    #如果是偶数个点，不处理，如果是奇数个点，处理两端的点（这步填充算法没有描述，我自己写的，不知道合不合理）
    if len(tmp_x)%2 != 0:
        head = 0
        tail = len(tmp_x)-1
        l = tail
        #处理向下的角
        if tmp_x[0][0] == tmp_x[1][0] and  tmp_x[0][1] !=  tmp_x[1][1]:
            head = 1
        if tmp_x[l][0] == tmp_x[l-1][0] and tmp_x[l][1] != tmp_x[l-1][1]:
            tail = l-1
    else:
        head = 0
        tail = len(tmp_x) - 1

    #开始染色
    if len(tmp_x) == 2:
        head = 0
        tail = 1
    if i == 50 :
        print(tmp_x)
    for k in range(head,tail,2):
        lx = tmp_x[k][0]
        rx = tmp_x[k+1][0]
        draw(lx,rx,i)






    # 遍历AET表，更新delx
    ptr = AET
    while ptr != None:
        #更新该活性边在下条扫描线对应的x值
        ptr.line['px']+=ptr.line['del_x']
        ptr = ptr.next


    #遍历AET表 删去y_max = i的点
    ptr = AET
    while ptr.next != None:
        nxt_ptr = ptr.next
        #删去这个结点
        if nxt_ptr.line['y_max'] == i:
            ptr.next = nxt_ptr.next
        else:
            if ptr.next != None:
                ptr = ptr.next




plt.show()
```
