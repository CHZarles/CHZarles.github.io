## 合并石头

传送门:[Acwing 合并石头](https://www.acwing.com/problem/content/description/284/)

```c++
#include<iostream>
#include<algorithm>
using namespace std;
int a[310][310];
int s[310];//前缀和数组
int main()
{
    int n;
    cin>>n;
    for(int i=1;i<=n;i++)
        {
            cin>>s[i];
            s[i]=s[i]+s[i-1];
        }
        
    //按长度计算状态
    for(int len=2;len<=n;len++)
    {
        //枚举起点
        for(int l=1;l+len-1<=n;l++)
        {
            //根据len和起点计算出终点
            int r = l + len - 1;
            
            //赋一个很大的初值
            a[l][r] = 1e9;
            //状态方程：a[l][r] = min( a[l][r] , a[l][k]+a[k+1][j]+s[r]-s[l-1])
            for(int k = l; k<r ; k++)
                a[l][r] = min(a[l][r],a[l][k]+a[k+1][r]+s[r]-s[l-1]);
        }   
    }    
    
    cout<<a[1][n];
}
```

![image-20201006105730639](https://gitee.com/chzarles/images/raw/master/imgs/image-20201006105730639.png)

```c++
#include<iostream>
#include <stdio.h>
#include<cstring>
using namespace std;
const int N = 120;
const int M = 300; 
char List[N];//存序列 
int Matrix[M][2];//存矩阵数字 
int n,m;
int a[N]; //存序列 

void solve()
{
	long long m[N][N];
    memset(m, 0, sizeof(m));
	
	for (int len = 2; len <= n; len++)           //区间长度
    	for (int i = 1; i + len - 1 <= n; i++) { //枚举起点
        	int j = i + len - 1;                 //区间终点
        	m[i][j] = 0x3f3f3f3f ;						//赋大值 
        	for (int k = i; k < j; k++) {        //枚举分割点，构造状态转移方程
            	m[i][j] = min(m[i][j], m[i][k] + m[k + 1][j] + a[i-1]*a[k]*a[j]);
        	}
    }
    
    cout<<m[1][n]<<endl;
    
}
int main()
{

   //freopen("input.txt", "r", stdin);

		char tmp;

		int flag = 0;

		}

		for(int i=0;i<n;i++)
			a[i] = Matrix[List[i]][0];
		a[n] =  Matrix[List[n-1]][1];

		solve() ;
```

## 最长公共子序列

题目：[最长公共子序列](https://www.acwing.com/problem/content/description/899/)

觉得这题的状态分成两半考虑比较方便，按两个序列末尾的字符是不是相等来区分。

![image-20201019090528211](https://gitee.com/chzarles/images/raw/master/imgs/image-20201019090528211.png)

如果两个字符相等，就可以直接转移到`f[i-1][j-1]`，不相等的话，两个字符一定有一个可以抛弃，可以对`f[i-1][j],f[i][j-1]`两种状态取max来转移。

![image-20201019090516629](https://gitee.com/chzarles/images/raw/master/imgs/image-20201019090516629.png)

```c++
#include <iostream>
using namespace std;
const int N = 1010;
int n, m;
char a[N], b[N];
int f[N][N];
int main() {
  cin >> n >> m >> a + 1 >> b + 1;
  for (int i = 1; i <= n; i++) {
    for (int j = 1; j <= m; j++) {
      if (a[i] == b[j]) {
        f[i][j] = f[i - 1][j - 1] + 1;
      } else {
        f[i][j] = max(f[i - 1][j], f[i][j - 1]);
      }
    }
  }
  cout << f[n][m] << '\n';
  return 0;
}
```

```c++
int Maxsum(int n, int *a)
{
	int res = 0 , b = 0;
	for(int i = 1 ; i <= n; i ++)  //枚举右边界
	{
		if(b > 0)
			b += a[i];
		else
			b = a[i];
		if(res < b)
			res = b;
	}
	return res;

}
```

**代码**

```c++
/*
n = x: 这个序列有x个像素值
s[i] = x: 压缩前i个像数值的最优解是x
p[k] = x: 第k个像素的灰度值是x
b[k] = x: 第k个像素的灰度值要用x位存
l[k] = x : 第k段有x个数(x<=256) 
*/
void Compress(int n, int p[], int s[], int l[], int b[])
{
	//每段最长有Lmax个像素值，存储段信息要用header位
	int Lmax = 256 , header = 11;
	
	//初始化s[0]
	s[0] = 0;
	
	//枚举像素列的结束点
	for( int i = 1 ; i <= n ; i ++)
	{
		//加入一个新像素点p[i],初始化各项信息
		b[i] = length(p[i]);
		int bmax = b[i];
		s[i] = s[i-1] + bmax;  //这里不急着加header
		l[i] = 1;
		

		for( int j = 2 ; j <= i && j <= Lmax; j ++)
		{
			if( bmax < b[i - j + 1] ) 
				bmax = b[i - j + 1];
			if( s[i] > s[i - j] + j * bmax )
				{
					s[i] = s[i - j] + j * bmax;
					l[i] = j;
				}
		}
		s[i] += header;
	}

}

int length(int i)
{
	int k = 1;
	i = i / 2 ; 
	while(i > 0)
	{
		k ++;
		i /= 2;
	}
	return k;	
}
```
