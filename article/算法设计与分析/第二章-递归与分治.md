## 整数划分



​	**将正整数n表示成一系列正整数之和**

```c++
#include<iostream>
using namespace std;
const int N = 10000;
int save[N], t;
/*
输出划分的结果
f(x,m) 表示将x分解成的式子最大的数不超过m的式子
递归方程：

		  f(x,x)      m>=x
 f(x,m) = m+f(x-m,m)  m<x
		   1          x=1

*/
int d;
void print_divide(int x, int m, int t)
{
	d++;
	if (m > x) m = x;
	if (x == 0)
	{
		cout << "=" << save[0];
		for (int i = 1; i < t; i++)
			cout << "+" << save[i];
		cout << endl;
		return;
	}
	for (int i = m; i >= 1; i--)
	{
		save[t++] = i;
		print_divide(x - i, i, t);
		t--; //恢复
	}
}


int k = 0;
int count_divide(int x,int m)
{
	k++; 
	if(m>x) return count_divide(x,x);
	if(x==1||m==1) return 1;  //只有一种情况 
	if(x==m) return 1 + count_divide(x,m-1);//x=m这种情况加count_divide(x,m-1)
	if(x>m&&m>1) return count_divide(x-m,m)+count_divide(x,m-1);//x分解成最大子整数是m的情况和最大子整数不是m的情况 
}
int main()
{
	print_divide(6, 6, 0);
	cout<<count_divide(6,6);
	
	cout<<"\ncount_divide()的递归次数: "<<k<<"\nprint_divide()的递归次数:  "<<d; 
	return 0;

}
```

## 棋盘覆盖问题

代码实现

```c++
//《算法设计与分析》上机：2.6 棋盘覆盖问题
 
#include <stdio.h>
 
#define N 16          //此处定义方格大小
int board[N][N]={0};  //表示棋盘
int tile=0;           //全局变量。表示L型骨牌的编号，初始值=0
 
void chessBoard(int tr,int tc,int dr,int dc,int size)
{
    if(size==1) return;
    int t;
    t=tile++;
    int s;
    s=size/2;
    //覆盖左上角
    if(dr<tr+s && dc<tc+s)
        chessBoard(tr,tc,dr,dc,s);
    else
    {
        board[tr+s-1][tc+s-1]=t; //左上角的右下角用骨牌覆盖，下面以此类推 
        chessBoard(tr,tc,tr+s-1,tc+s-1,s);
    }
    //覆盖右上角
    if(dr<tr+s && dc>=tc+s)
        chessBoard(tr,tc+s,dr,dc,s);
    else
    {
        board[tr+s-1][tc+s]=t;
        chessBoard(tr,tc+s,tr+s-1,tc+s,s);
    }
    //覆盖左下角
    if(dr>=tr+s && dc<tc+s)
        chessBoard(tr+s,tc,dr,dc,s);
    else
    {
        board[tr+s][tc+s-1]=t;
        chessBoard(tr+s,tc,tr+s,tc+s-1,s);
    }
    //覆盖右下角
    if(dr>=tr+s && dc>=tc+s)
        chessBoard(tr+s,tc+s,dr,dc,s);
    else
    {
        board[tr+s][tc+s]=t;
        chessBoard(tr+s,tc+s,tr+s,tc+s,s);
    }
}
 
void display()
{
    int i,j;
    for(i=0;i<N;i++)
    {
        for(j=0;j<N;j++)
            printf("%2d ",board[i][j]);
        printf("\n");
    }
}
int main()
{
    int tr=0,tc=0;//tr,tc分别表示左上角方格的行号和列号
    chessBoard(tr,tc,1,10,N);//第一行第10列是不可覆盖快 
    display();
    return 0;
}
```









## 线性时间选择算法

**Q:给定线性序集中n个元素和一个整数k，1≤k≤n，要求找出这n个元素中第k小的元素**

**步骤:**

`Select函数`功能:

- Step 1：将n个输入元素划分成n/5（向上取整）个组，每组5个元素(最多只可能有一个组不是5个元素)。用任意一种排序算法，将每组中的元素排好序，并取出每组的中位数，共n/5（向上取整）个。 
- Step 2:  递归调用`select函数`来找出这n/5（向上取整）个元素的中位数。如果n/5（向上取整）是偶数，就找它的2个中位数中较大的一个。以这个元素作为划分基准,调用 `Partition函数`,将序列分成两部分，取需要的部分，再运行`Select函数`。

![image-20200921172359740](https://gitee.com/chzarles/images/raw/master/imgs/image-20200921172359740.png)

> 有5个数的组数：n/5 - 1
>
> 设中位数的中位数是x，比x小和比x大的元素至少 3\*(n/5-1)/2  => 3\*(n-5)/10
>
> tip: (n/5-1)/2 是组数，3是每组至少有三个比紫色点小或大
>
> 这个算法无论什么情况都可以在线性时间找到第k小的数



```c++
//主要步骤就是分组，求中位数，求中位数的中位数x，根据x再重新划分数组，如此递归
#include<iostream>
using namespace std;
int a[100010],n,k;

void Sort(int a[],int l,int r)
{
	for(int i=l; i<r; i++)
	{
		//确定i指向的地方的值 
		for(int j=i+1; j<=r; j++)
		{
			//升序排序 
			if(a[j]<a[i])swap(a[i],a[j]);
		}
	}
}

// 划分 
int Partition(int q[],int l,int r,int val)
{
	if(l>=r) return r;
    int i = l - 1 , j = r + 1;
    while(i < j)
    {
        do i++; while(q[i]<val);
        do j--; while(q[j]>val);
        if(i<j) swap(q[i],q[j]);
    }
    return j;
}

//左闭右闭 
int Select(int a[],int l,int r,int k)
{
	//n<75 规模变小了直接调用sort
	if(r-l<75)
	{
		Sort(a,l,r);
		return a[l+k-1];
	}
	
	
	//找中位数的中位数，r-p-4即上面所说的n-5
	//这里的group_nums不是准确的组数 
	int group_nums =  (r-l-4)/5;
	for(int i=0; i<=group_nums; i++) //把每个组的中位数交换到区间[l,l+(r-l-4)/4]
	{
		//第i组开头的索引 
		int s=l+5*i;
		//对第i组排序 
		Sort(a,s,s+4);
		//交换每组中的中位数到数组a[l+0]~a[l+group_nums],方便后续处理 
		swap(a[l+i],a[s+2]);
	}
	
	//求中位数的中位数，共有group_nums+1个中位数
	//左闭右闭 
	int x=Select(a,l,l+group_nums,(group_nums+1)/2);
	
	//以x的值为基点进行划分,返回划分后x所在的索引i 
	int i=Partition(a,l,r,x);
	//求出划分后左边数组的长度 
	int left_len=i-l+1;
	//要求的数在左边的数组中 
	if(k<=left_len) return Select(a,l,i,k);
	else return Select(a,i+1,r,k-left_len);
}

int main()
{
    cin>>n>>k;
    for(int i = 0 ; i <n ; i++ )
    cin>>a[i];
    Select(a,0,n-1,k);
    cout<<a[k-1];
    
}
```





## 最近点对问题

最近点对问题的变体:[Quoit Design](http://acm.hdu.edu.cn/showproblem.php?pid=1007)

`Solve函数`

- **参数**: 数组`pt[]` 的左右界限索引`l`,`r`（数组`pt[i]`存储点的信息，且`pt[i+1].x>=p[i].x ,pt[i+1].x==p[i].x时,pt[i+1].y>=p[i].y`）

  

- **出口**: 还剩小于三个点，暴力求解最小距离，然后返回最小点对距离值



- **主过程**: 

  - **Step1** 把数组二分成两份，`pt[m]`(m = l+r>>1) 是划分的基点，调用Solve函数求出两个区间各自的最小点对距离，并选出其中最小的，记为`D`

  - **Step2**  找出所有`x坐标`在区间`[pt[m].x-D,pt[m].x+D]`的点,存到数组`a[]`中，对`a[]`排序，然后遍历`a[]`暴力计算出每两个点间的距离。

  - **Step3** 最后返回最优解

    

    > Step2的暴力的复杂度不是O(N^2/4) ，而是O(3N) . 因为以下前提条件限制了搜索范围
    >
    > ​    说明: 我们把区间[`[pt[m].x-D,pt[m].x]` 称为左半区间，`[pt[m].x,pt[m].x+D]`称为右区间 
    >
    > ​	条件1：左(右)半区间里的点之间的距离都小于D  =>  在一个区间里任意一个D*2D的矩形空间最多只有六个点。
    >
    > ​    条件2：对于左半(右半)区间里的任意一个点`p0`，要在另一半找点`p1` ，令`distance(p0,p1) <=D` ,只需要在与自己临近（**什么叫临近? 看下面的图**）的矩形空间找。
    >
    > 
    >
    >  **所以结论就是，每个点只要算6次就保证一定能找到最优解，复杂度O(N/2*6) = O(3N)**

![image-20200927160209673](https://gitee.com/chzarles/images/raw/master/imgs/image-20200927160209673.png)

```c++
#include <cstdio>
#include <algorithm>
#include <cmath>
using namespace std;
#define N 100007

using namespace std;

//存点 
struct Point
{
    double x,y;
}pt[N];
int a[N];

int n;
//按x升序排序，y升序排序 
bool cmp(Point a, Point b)
{
    if (a.x != b.x) return a.x < b.x;
    else return a.y < b.y;
}
//升序排序 
bool cmp_y(int id1, int id2){
    return pt[id1].y < pt[id2].y;
}

//计算距离 
double getDis(const Point &a, const Point &b)
{
   double x = a.x - b.x;
   double y = a.y - b.y;
   return sqrt(x*x + y*y);
}


double solve(int l, int r)
{
    double D = 0;
    //递归出口: 如果整个区间小于3个点 
    if (r - l + 1 <= 3)
    {
    	//如果只剩一个点,d = 0,所以return 0 
        if (r - l + 1 == 1) return D;
        
        //计算第一对点 
        D = getDis(pt[l], pt[l + 1]);
        
        //如果只剩两个点，返回刚才的计算值 
        if (r - l + 1 == 2) return D;
        
        //还剩三个点的情况 
        for (int i = l; i < r; ++i)
        {
        	//两两求距离，并取最小值返回 
            for (int j = i + 1; j <= r; ++j)
            {
                D = min(D, getDis(pt[i],pt[j]));
            }
        }
        return D;
    }
    
    //把点分成两份 
    int m = (l + r) >> 1;
    double s1 = solve(l, m);
    double s2 = solve(m + 1,  r);
    //取出其中的d_min 
    D = min(s1,s2);
    
    
    int k = 0;
    //用到了课本上的思想，求区间 [pt[m].x-m,pt[x].x+m] 的点
	//并把这些点的索引存到数组a上 
    for (int i = m - 1; i >= l && pt[m].x - pt[i].x <= D; --i) a[k++] = i;
    for (int i = m + 1; i <= r && pt[i].x - pt[m].x <= D; ++i) a[k++] = i;

    
    //对这些点按x升序,y升序排序，保证能找相邻的6个点 
    sort(a, a + k, cmp_y);
    
    //两两计算点对
	//求出距离最小值 
    //有计算同区间点对的嫌疑，但无伤大雅
    for (int i = 0; i < k; ++i)
    {
        for (int j = i + 1; j < k && j <= i + 7; ++j)
        {
            D = min(D, getDis(pt[a[i]], pt[a[j]]));
        }
    }
    return D;
}



int main()
{
	//n个点 
    while (~scanf("%d",&n))
    {
        if (!n) break;
        //储存每个点的坐标 
        for (int i = 0; i < n; ++i) scanf("%lf%lf",&pt[i].x, &pt[i].y);
        //对点升序排序 
        sort(pt, pt + n, cmp);
        printf("%.2lf\n",solve(0, n - 1)/2.0);
    }
    return 0;
}
```
